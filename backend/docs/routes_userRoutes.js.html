/**
 * @file userRoutes.js
 * @description Routes Express pour la gestion des utilisateurs (CRUD + Login + Logout)
 */

const express = require("express");
const router = express.Router();
const {
  getUsers,
  getUserByEmail,
  createUser,
  updateUser,
  deleteUser,
  loginUser,
  logoutUser,
} = require("../controllers/userController");

const auth = require("../middleware/auth");

/**
 * @route POST /users/
 * @description Crée un nouvel utilisateur
 * @access Public
 */
router.post("/", createUser);

/**
 * @route GET /users/
 * @description Liste tous les utilisateurs
 * @access Private
 */
router.get("/", auth, getUsers);

/**
 * @route GET /users/:email
 * @description Récupère un utilisateur par son email
 * @access Private
 */
router.get("/:email", auth, getUserByEmail);

/**
 * @route PUT /users/:email
 * @description Modifie un utilisateur
 * @access Private
 */
router.put("/:email", auth, updateUser);

/**
 * @route DELETE /users/:email
 * @description Supprime un utilisateur
 * @access Private
 */
router.delete("/:email", auth, deleteUser);

/**
 * @route POST /login
 * @description Connexion utilisateur + renvoi d’un token JWT
 * @access Public
 */
router.post("/login", loginUser);

/**
 * @route GET /logout
 * @description Déconnexion utilisateur
 * @access Public
 */
router.get("/logout", logoutUser);

module.exports = router;
